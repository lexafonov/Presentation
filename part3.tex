				\nu_2 \sigma + \delta_2 R_1^{\frac12} + R_1 + 2 \gamma_{02} R_2 &= 0.
			\end{aligned}
		\end{equation*}

					\end{aligned}
				\end{align*}
			}

		$\nu_1 = 0$, $\nu_2 \neq 0$

			\begin{align*}

					\nu_2 \sigma + \delta_2 R_1^{\frac12} + R_1 + 2 \gamma_{02} R_2 &= 0.
				\end{aligned}
			\end{align*}
		}

	\begin{frame}%Постановка задачи:
		\frametitle{ I. Случай точного резонанса. Постановка задачи:}
		Неавтономная 2$\pi$-периодическая по времени гамильтонова система с двумя степенями свободы
		\begin{multline*}
			H\skobkiKr{q_j, p_j, t} = \cfrac12 \lambda_1 \skobkiKr{q_1^2 + p_1^2} + \cfrac12 \sigma \lambda_2 \skobkiKr{q_2^2 + p_2^2} + H_3\skobkiKr{q_j, p_j, t} +\\
			+H_4\skobkiKr{q_j, p_j, t} +O_5, \quad \skobkiKr{j=1,2}
		\end{multline*}
		Назовем резонанс сильным, если он может привести к неустойчивости в системе, и слабым, если его наличие не приводит к неустойчивости.
		
		Назовем резонанс основным, если в резонансном соотношении 
		\begin{align*}
			m_1 \lambda_1 + m_2 \lambda_2 = l
		\end{align*}
		присутствует только одна из величин $\lambda_j$, и комбинационным, если в соотношении имеются обе величины $\lambda_1$ и $\lambda_2$.
	\end{frame}

	\begin{frame}%Постановка задачи:
		\frametitle{ I. Постановка задачи:}
		Пусть величины $\lambda_1$ и $\lambda_2$ задаются одним из наборов ($k_1$, $k_2$ — целые)
		\begin{columns}
			\column{0.5\textwidth}
				\begin{align*}
					\lambda_1 &= k_1 + 1/3, & \lambda_2 &= k_2 + 5/6,\\
					\lambda_1 &= k_1 + 1/3, & \lambda_2 &= k_2 + 1/6,\\
				\end{align*}
			\column{0.5\textwidth}
				\begin{align*}
					\lambda_1 &= k_1 + 2/3, & \lambda_2 &= k_2 + 1/6,\\
					\lambda_1 &= k_1 + 2/3, & \lambda_2 &= k_2 + 5/6.\\
				\end{align*}
		\end{columns}
		
		Величины $3\lambda_1$, $\lambda_1 - 2\lambda_2$ и $3\lambda_1$, $\lambda_1 + 2\lambda_2$ являются целыми.\hfill \break
		
		Резонансные коэффициенты малы и имеют порядок $\varepsilon$ $\skobkiKr{0 < \varepsilon \ll 1}$.\hfill \break
		
		{\bf Цель данной работы} — упрощение гамильтониана, нахождение положений равновесия модельной системы и условий их существования, рассмотрение вопроса об устойчивости положений равновесия и соответствующих им движений полной системы в $\varepsilon$-окрестности тривиального положения равновесия.
	\end{frame}

		\frametitle{ II. Случай, близкий к точному резонансу. Постановка задачи:}
		\begin{enumerate}
			\item Проведем аналогичную серию канонических замен для упрощения вида гамильтониана вомущенного движения. 
			
			\item Введем резонансные расстройки $\mu_j$ по формулам $\lambda_j = \lambda_{j0} - \varepsilon^2 \mu_j$ $(j = 1,2)$. 
			
			\item Перейдем к новому времени $\tau=\varepsilon^2 t$ 
		\end{enumerate}
		\begin{equation*}\label{Ham1R}
			\begin{aligned}
				\Gamma_{\pm}^* = \nu_1 R_1 \pm \nu_2 R_2 + \alpha R_1^{\frac32} \cos 3 \Phi_1 &+ R_1^{\frac12} R_2 \cos \left( \Phi_1 \pm 2 \Phi_2 \right) +\\
				&+\gamma_{20} R_1^2 + R_1 R_2 + \gamma_{02} R_2^2 + O(\varepsilon),
			\end{aligned}
		\end{equation*}
		где $\nu_j$ - коэффициенты, характеризующие резонансные расстройки, а остальные коэффициенты аналогичны I.

		Равновесные значения величин $R_1$ и $R_2$ задаются системой двух алгебраических уравнений
		
		\begin{equation*}\label{SuR}
			\begin{aligned}
				\nu_1 + \frac32 \alpha \delta_1 R_1^{\frac12} + \frac12 \delta_2 R_1^{-\frac12} R_2 + 2 \gamma_{20} R_1 + R_2&= 0,\\
				\nu_2 \sigma + \delta_2 R_1^{\frac12} + R_1 + 2 \gamma_{02} R_2 &= 0.
			\end{aligned}
		\end{equation*}
		\hfill \break
		
		\begin{tcolorbox}
			$\nu_1 \neq 0$, $\nu_2 = 0$
			{
				\begin{align*} 
					\begin{aligned}
						\nu_1 + \frac32 \alpha \delta_1 R_1^{\frac12} + \frac12 \delta_2 R_1^{-\frac12} R_2 + 2 \gamma_{20} R_1 + R_2&= 0,\\
						\delta_2 R_1^{\frac12} + R_1 + 2 \gamma_{02} R_2 &= 0.
					\end{aligned}
				\end{align*}
			}
		\end{tcolorbox}
		
		$\nu_1 = 0$, $\nu_2 \neq 0$
		{
			\begin{align*} 
				\begin{aligned}
					\frac32 \alpha \delta_1 R_1^{\frac12} + \frac12 \delta_2 R_1^{-\frac12} R_2 + 2 \gamma_{20} R_1 + R_2&= 0,\\
					\nu_2 \sigma + \delta_2 R_1^{\frac12} + R_1 + 2 \gamma_{02} R_2 &= 0.
				\end{aligned}
			\end{align*}
		}
